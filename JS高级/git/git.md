git add 提交到暂存区
git status 查看暂存区状态
git commit -m "提交描述"
git config --global user.name '1205599524' 设置用户名
git config --global user.eamil '1205599524@qq.com' 设置用户邮箱
在文件内初始化项目 git init
git rm 从暂存区删除文件
git clone 仓库地址 克隆项目
git config --list 查看配置
git push 
git branch -a 查看分支
git branch 查看当前分支 创建分支
git checkout 分支名 切换分支
git branch -d test 删除分支
